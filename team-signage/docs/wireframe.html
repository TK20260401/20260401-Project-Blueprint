function doGet(e) {
  const ss = SpreadsheetApp
    .getActiveSpreadsheet();
  const sheet = ss.getSheetByName("在席管理");
  const data = sheet.getDataRange().getValues();
  const headers = data.shift();
  const result = data.map(row => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = row[i]);
    return obj;
  });
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const sheet = SpreadsheetApp
    .getActiveSpreadsheet()
    .getSheetByName("営業カレンダー");
  const today = new Date();
  const endDate = new Date(
    today.getTime() + 7*24*60*60*1000
  );
  // 今日〜7日後のデータをフィルタ
  const filtered = allData.filter(
    row => row.date >= today
      && row.date <= endDate
  );
  return ContentService
    .createTextOutput(
      JSON.stringify(filtered)
    ).setMimeType(ContentService.MimeType.JSON);
}

データスキーマ定義

3つのGoogle Spreadsheetシートのカラム定義です。 すべてのシートは同一スプレッドシートファイル内に配置します。

📊 シート1: 在席管理
カラム名 型 必須 説明 例
name 文字列 必須 メンバー氏名 共栄 太郎
status 文字列 (enum) 必須 在席ステータス 在席 / 外出 / 会議中 / 休み / 接客 / 車
location 文字列 任意 現在地・場所 本社 / ○○商事 / 会議室A
work_type 文字列 任意 業務種別 デスクワーク / 訪問 / 打ち合わせ
memo 文字列 任意 一行メモ（自由記述） ○○案件→▲▲案件 直帰
updated_at 日時 自動 最終更新日時 2023-01-10T09:02:00
📊 シート2: 営業カレンダー
カラム名 型 必須 説明 例
date 日付 必須 予定日 2023-01-10
name 文字列 必須 担当者名 田中 花子
content 文字列 必須 予定内容 ○○商事 訪問（新規提案）
type 文字列 (enum) 任意 予定種別 外出 / 会議 / 研修 / 休暇
location 文字列 任意 訪問先・場所 ○○商事本社
📊 シート3: リソース管理
カラム名 型 必須 説明 例
project_name 文字列 必須 案件・現場名 ○○ビル新築工事
assigned_members 文字列 (CSV) 必須 担当メンバー（カンマ区切り） 共栄太郎,鈴木健一,山田大輔

認証・アクセス制御

本システムはログイン不要の構成です。GAS Web Appをパブリック公開し、 データアクセスはSpreadsheetレベルの権限で制御します。

🔒
認証方式: ログイン不要（パブリックGAS Web App）

GAS Web App公開設定:
・実行ユーザー: 「自分」（スプレッドシートオーナー）
・アクセス権限: 「全員（匿名ユーザーを含む）」

セキュリティ対策:
・Google Spreadsheetの共有設定で閲覧権限を管理
・GAS Web AppのURL自体が認証トークンの代替
・社内ネットワーク内のサイネージ端末のみで使用を想定
・書き込みAPIは公開しない（読み取り専用: doGet のみ）

推奨構成:
・サイネージ端末は社内LAN接続の専用デバイス
・Spreadsheetの編集権限はチームメンバーのみに付与
・GAS URLは外部に漏洩しないよう管理

// 表示切替ロジック（自動ローテーション）
const screens = [
  { id: "status-board", duration: 30000 },
  { id: "calendar", duration: 20000 },
  { id: "capacity", duration: 20000 }
];
let currentIdx = 0;

function rotateScreen() {
  screens.forEach((s, i) => {
    document.getElementById(s.id)
      .style.display = i === currentIdx
        ? "block" : "none";
  });
  setTimeout(() => {
    currentIdx = (currentIdx + 1) % screens.length;
    rotateScreen();
  }, screens[currentIdx].duration);
}
rotateScreen();

// GAS Web App URL（デプロイ後に取得）
const GAS_BASE = "https://script.google.com/macros/s/XXXX/exec";

// 在席管理: 30秒ポーリング
async function fetchStatus() {
  try {
    const res = await fetch(
      GAS_BASE + "?action=status"
    );
    const data = await res.json();
    renderStatusBoard(data);
  } catch(e) {
    console.error("Status fetch failed", e);
  }
}
setInterval(fetchStatus, 30000);
