// 稼働率 = 実績工数 / 計画工数 × 100 const utilization = (actual_hours / planned_capacity) * 100; // チーム稼働率 = Σ個人稼働率 / メンバー数 const teamRate = members.reduce((s, m) => s + m.util, 0) / members.length; // 空きリソース = 稼働率50%未満のメンバー数 const available = members.filter(m => m.util < 50).length;
